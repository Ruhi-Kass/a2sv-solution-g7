from typing import List

class Solution:
    def findRelativeRanks(self, score: List[int]) -> List[str]:
        n = len(score)
        
      
        athletes = [(score[i], i) for i in range(n)]
        
      
        athletes.sort(reverse=True, key=lambda x: x[0])
        
     
        result = [""] * n
        
        for rank in range(n):
            original_idx = athletes[rank][1]
            if rank == 0:
                result[original_idx] = "Gold Medal"
            elif rank == 1:
                result[original_idx] = "Silver Medal"
            elif rank == 2:
                result[original_idx] = "Bronze Medal"
            else:
                result[original_idx] = str(rank + 1)
        
        return result
